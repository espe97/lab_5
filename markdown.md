## –¶–µ–ª—å —Ä–∞–±–æ—Ç—ã
–†–µ–∞–ª–∏–∑–æ–≤–∞—Ç—å –∏ –ø—Ä–æ—Ç–µ—Å—Ç–∏—Ä–æ–≤–∞—Ç—å –≤–µ—Ä–æ—è—Ç–Ω–æ—Å—Ç–Ω—ã–µ –∞–ª–≥–æ—Ä–∏—Ç–º—ã –ø—Ä–æ–≤–µ—Ä–∫–∏ —á–∏—Å–µ–ª –Ω–∞ –ø—Ä–æ—Å—Ç–æ—Ç—É:
1. –ê–ª–≥–æ—Ä–∏—Ç–º, —Ä–µ–∞–ª–∏–∑—É—é—â–∏–π —Ç–µ—Å—Ç –§–µ—Ä–º–∞.
2. –ê–ª–≥–æ—Ä–∏—Ç–º –≤—ã—á–∏—Å–ª–µ–Ω–∏—è —Å–∏–º–≤–æ–ª–∞ –Ø–∫–æ–±–∏.
3. –ê–ª–≥–æ—Ä–∏—Ç–º, —Ä–µ–∞–ª–∏–∑—É—é—â–∏–π —Ç–µ—Å—Ç –°–æ–ª–æ–≤–µ—è-–®—Ç—Ä–∞—Å—Å–µ–Ω–∞.
4. –ê–ª–≥–æ—Ä–∏—Ç–º, —Ä–µ–∞–ª–∏–∑—É—é—â–∏–π —Ç–µ—Å—Ç –ú–∏–ª–ª–µ—Ä–∞-–†–∞–±–∏–Ω–∞.

–≠—Ç–∏ –∞–ª–≥–æ—Ä–∏—Ç–º—ã –ø–æ–∑–≤–æ–ª—è—é—Ç –æ–ø—Ä–µ–¥–µ–ª–∏—Ç—å, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ –∑–∞–¥–∞–Ω–Ω–æ–µ —á–∏—Å–ª–æ **–≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç—ã–º** –∏–ª–∏ **—Å–æ—Å—Ç–∞–≤–Ω—ã–º** —Å –≤—ã—Å–æ–∫–æ–π —Å—Ç–µ–ø–µ–Ω—å—é –Ω–∞–¥—ë–∂–Ω–æ—Å—Ç–∏.

## 1. –ê–ª–≥–æ—Ä–∏—Ç–º, —Ä–µ–∞–ª–∏–∑—É—é—â–∏–π —Ç–µ—Å—Ç –§–µ—Ä–º–∞

**–û–ø–∏—Å–∞–Ω–∏–µ:** –¢–µ—Å—Ç –§–µ—Ä–º–∞ –æ—Å–Ω–æ–≤–∞–Ω –Ω–∞ **–º–∞–ª–æ–π —Ç–µ–æ—Ä–µ–º–µ –§–µ—Ä–º–∞**. –ï—Å–ª–∏ \( p \) ‚Äî –ø—Ä–æ—Å—Ç–æ–µ —á–∏—Å–ª–æ, –∏ \( a \) ‚Äî –ª—é–±–æ–µ —Ü–µ–ª–æ–µ —á–∏—Å–ª–æ, —Ç–∞–∫–æ–µ —á—Ç–æ \( 1 < a < p-1 \), —Ç–æ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è —Å–ª–µ–¥—É—é—â–µ–µ —Å—Ä–∞–≤–Ω–µ–Ω–∏–µ:
\[
a^{p-1} \equiv 1 \ (\text{mod} \ p)
\]
–ï—Å–ª–∏ —ç—Ç–æ —É—Å–ª–æ–≤–∏–µ –Ω–µ –≤—ã–ø–æ–ª–Ω—è–µ—Ç—Å—è –¥–ª—è —Å–ª—É—á–∞–π–Ω–æ–≥–æ –∑–Ω–∞—á–µ–Ω–∏—è \( a \), —Ç–æ \( n \) —è–≤–ª—è–µ—Ç—Å—è —Å–æ—Å—Ç–∞–≤–Ω—ã–º.

**–†–µ–∞–ª–∏–∑–∞—Ü–∏—è –Ω–∞ Python:**

```python
import random

def fermat_primality_test(n, k=5):
    """–ü—Ä–æ–≤–µ—Ä—è–µ—Ç, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —á–∏—Å–ª–æ n –≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç—ã–º —Å –ø–æ–º–æ—â—å—é —Ç–µ—Å—Ç–∞ –§–µ—Ä–º–∞."""
    if n <= 1:
        return False
    if n <= 3:
        return True
    for _ in range(k):
        a = random.randint(2, n - 2)
        if pow(a, n - 1, n) != 1:
            return False  # n —Å–æ—Å—Ç–∞–≤–Ω–æ–µ
    return True  # n –≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç–æ–µ
```
# 2. –ê–ª–≥–æ—Ä–∏—Ç–º –≤—ã—á–∏—Å–ª–µ–Ω–∏—è —Å–∏–º–≤–æ–ª–∞ –Ø–∫–æ–±–∏

```python
def jacobi_symbol(a, n):
    """–í—ã—á–∏—Å–ª–µ–Ω–∏–µ —Å–∏–º–≤–æ–ª–∞ –Ø–∫–æ–±–∏ (a/n)."""
    result = 1
    while a != 0:
        while a % 2 == 0:
            a //= 2
            if n % 8 in [3, 5]:
                result = -result
        a, n = n, a  # –ú–µ–Ω—è–µ–º a –∏ n –º–µ—Å—Ç–∞–º–∏
        if a % 4 == 3 and n % 4 == 3:
            result = -result
        a %= n
    return result if n == 1 else 0

```

# 3. –ê–ª–≥–æ—Ä–∏—Ç–º, —Ä–µ–∞–ª–∏–∑—É—é—â–∏–π —Ç–µ—Å—Ç –°–æ–ª–æ–≤–µ—è-–®—Ç—Ä–∞—Å—Å–µ–Ω–∞
–û–ø–∏—Å–∞–Ω–∏–µ: –¢–µ—Å—Ç –°–æ–ª–æ–≤–µ—è-–®—Ç—Ä–∞—Å—Å–µ–Ω–∞ –ø—Ä–æ–≤–µ—Ä—è–µ—Ç –ø—Ä–æ—Å—Ç–æ—Ç—É —á–∏—Å–ª–∞, –∏—Å–ø–æ–ª—å–∑—É—è —Å–∏–º–≤–æ–ª –Ø–∫–æ–±–∏ –∏ –∫—Ä–∏—Ç–µ—Ä–∏–π –≠–π–ª–µ—Ä–∞.

```python
def solovay_strassen_test(n, k=5):
    """–ü—Ä–æ–≤–µ—Ä—è–µ—Ç, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —á–∏—Å–ª–æ n –≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç—ã–º —Å –ø–æ–º–æ—â—å—é —Ç–µ—Å—Ç–∞ –°–æ–ª–æ–≤–µ—è-–®—Ç—Ä–∞—Å—Å–µ–Ω–∞."""
    if n < 2:
        return False
    if n != 2 and n % 2 == 0:
        return False
    for _ in range(k):
        a = random.randint(2, n - 2)
        x = jacobi_symbol(a, n)
        if x == 0 or pow(a, (n - 1) // 2, n) != (x % n):
            return False  # n —Å–æ—Å—Ç–∞–≤–Ω–æ–µ
    return True  # n –≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç–æ–µ

```

# 4. –ê–ª–≥–æ—Ä–∏—Ç–º, —Ä–µ–∞–ª–∏–∑—É—é—â–∏–π —Ç–µ—Å—Ç –ú–∏–ª–ª–µ—Ä–∞-–†–∞–±–∏–Ω–∞

```python
def miller_rabin_primality_test(n, k=5):
    """–ü—Ä–æ–≤–µ—Ä—è–µ—Ç, —è–≤–ª—è–µ—Ç—Å—è –ª–∏ —á–∏—Å–ª–æ n –≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç—ã–º —Å –ø–æ–º–æ—â—å—é —Ç–µ—Å—Ç–∞ –ú–∏–ª–ª–µ—Ä–∞-–†–∞–±–∏–Ω–∞."""
    if n <= 1:
        return False
    if n <= 3:
        return True
    if n % 2 == 0:
        return False

    # –†–∞–∑–ª–æ–∂–µ–Ω–∏–µ n - 1 –≤ –≤–∏–¥ 2^s * d
    s = 0
    d = n - 1
    while d % 2 == 0:
        d //= 2
        s += 1

    for _ in range(k):
        a = random.randint(2, n - 2)
        x = pow(a, d, n)
        if x == 1 or x == n - 1:
            continue
        for _ in range(s - 1):
            x = pow(x, 2, n)
            if x == n - 1:
                break
        else:
            return False  # n —Å–æ—Å—Ç–∞–≤–Ω–æ–µ
    return True  # n –≤–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç–æ–µ
```

# –ü—Ä–∏–º–µ—Ä –∏—Å–ø–æ–ª—å–∑–æ–≤–∞–Ω–∏—è
–î–∞–Ω–Ω—ã–π –ø—Ä–∏–º–µ—Ä –ø–æ–∑–≤–æ–ª—è–µ—Ç –ø–æ–ª—å–∑–æ–≤–∞—Ç–µ–ª—é –≤–≤–æ–¥–∏—Ç—å —á–∏—Å–ª–æ n , –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∏—Ç–µ—Ä–∞—Ü–∏–π ùëò –¥–ª—è –ø–æ–≤—ã—à–µ–Ω–∏—è –Ω–∞–¥—ë–∂–Ω–æ—Å—Ç–∏, –∏ –∑–Ω–∞—á–µ–Ω–∏–µ a –¥–ª—è –≤—ã—á–∏—Å–ª–µ–Ω–∏—è —Å–∏–º–≤–æ–ª–∞ –Ø–∫–æ–±–∏.

k = int(input("–í–≤–µ–¥–∏—Ç–µ –∫–æ–ª–∏—á–µ—Å—Ç–≤–æ –∏—Ç–µ—Ä–∞—Ü–∏–π –¥–ª—è –ø–æ–≤—ã—à–µ–Ω–∏—è –Ω–∞–¥—ë–∂–Ω–æ—Å—Ç–∏: "))

print("\n–†–µ–∑—É–ª—å—Ç–∞—Ç—ã –ø—Ä–æ–≤–µ—Ä–∫–∏ —á–∏—Å–ª–∞", n, "–Ω–∞ –ø—Ä–æ—Å—Ç–æ—Ç—É:")

# –°–∏–º–≤–æ–ª –Ø–∫–æ–±–∏ –≤—ã—á–∏—Å–ª—è–µ—Ç—Å—è –¥–ª—è –ø—Ä–∏–º–µ—Ä–∞ —Å a = 2, –º–æ–∂–Ω–æ –∏–∑–º–µ–Ω–∏—Ç—å –Ω–∞ –¥—Ä—É–≥–æ–µ –∑–Ω–∞—á–µ–Ω–∏–µ

print("–¢–µ—Å—Ç –°–æ–ª–æ–≤–µ—è-–®—Ç—Ä–∞—Å—Å–µ–Ω–∞:", "–í–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç–æ–µ" if solovay_strassen_test(n, k) else "–°–æ—Å—Ç–∞–≤–Ω–æ–µ")
print("–¢–µ—Å—Ç –ú–∏–ª–ª–µ—Ä–∞-–†–∞–±–∏–Ω–∞:", "–í–µ—Ä–æ—è—Ç–Ω–æ –ø—Ä–æ—Å—Ç–æ–µ" if miller_rabin_primality_test(n, k) else "–°–æ—Å—Ç–∞–≤–Ω–æ–µ")
